    # The compress function collapses repeated directories into # a single entry with a time-stamp yielding an equivalent probability. _scd_Y19oug_compress() {

    awk -v epochseconds=$EPOCHSECONDS \
    -v meanlife=$SCD_MEANLIFE \
    -v minlogprob=$MINLOGPROB \
    '
    BEGIN {
        FS = "[:;]";
        pmin = exp(minlogprob);
    }
    /^: deleted:0;/ { next; }
    length($0) < 4096 && $2 > 1000 {
        df = $0;
        sub("^[^;]*;", "", df);
        if (!df)  next;
        tau = 1.0 * ($2 - epochseconds) / meanlife;
        prob = (tau < minlogprob) ? pmin : exp(tau);
        dlist[last[df]] = "";
        dlist[NR] = df;
        last[df] = NR;
        ptot[df] += prob;
    }
    END {
        for (i = 1; i <= NR; ++i) {
            d = dlist[i];
            if (d) {
                ts = log(ptot[d]) * meanlife + epochseconds;
                printf(": %.0f:0;%s\n", ts, d);
            }
        }
    }
    ' $*

    }
